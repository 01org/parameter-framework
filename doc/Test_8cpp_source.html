207  // Build a criterion with as many value as there is bits in int.
208  std::vector<char> names(sizeof(int) * CHAR_BIT + 1, 'a');
209  names.back() = '\0';
210  std::vector<const char *> values(names.size());
211  for(size_t i = 0; i < values.size(); ++i) {
212  values[i] = &names[i];
213  }
214  values.back() = NULL;
215  /* The pfw c api requires criterion values to be a NULL terminated
216  * array of string. Each string is a pointer to a NULL terminated
217  * array of char. The pfw requires each string to be different
218  * from all others, ie strcmp(values[i], values[j]) != 0 for any
219  * i j.
220  *
221  * In order to generate easily an array of different strings,
222  * instantiate one string (names) big enough
223  * (@see PfwCriterion::values).
224  * Then instantiate an array of pointer (values),
225  * each pointing to a different position in the previously
226  * created string.
227  *
228  * Representation of the names and values vectors.
229  *
230  * n = number of bit in an int
231  * <--- n+1 elements --->
232  * names = |a|a|a|a|...|a|a|a|\0|
233  * ^ ^ ^
234  * values[0] = ´ | |
235  * values[1] = --´ |
236  * ... |
237  * values[n - 1] = -----------´
238  * values[n] = NULL
239  *
240  */
241  const PfwCriterion duplicatedCriteria[] = {{"name", true, &values[0]}};
