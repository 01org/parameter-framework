194 template <class T>
195 bool ElementHandle::setAs(const T value, string &error) const
196 {
197  if (not checkSetValidity(getSize(value), error)) {
198  return false;
199  }
200  // Safe downcast thanks to isParameter check in checkSetValidity
201  auto &parameter = static_cast<CBaseParameter &>(mElement);
202 
203  // When in tuning mode, silently skip "set" requests
204  if (mParameterMgr.tuningModeOn()) {
205 
206  return true;
207  }
208 
209  CParameterAccessContext parameterAccessContext(error, mParameterMgr.getParameterBlackboard());
210 
211  // BaseParamere::access takes a non-const argument - therefore we need to
212  // copy the value
213  T copy = value;
214 
215  // Ensure we're safe against blackboard foreign access
216  lock_guard<mutex> autoLock(mParameterMgr.getBlackboardMutex());
217 
218  return parameter.access(copy, true, parameterAccessContext);
219 }

343 bool ElementHandle::checkGetValidity(bool asArray, string &error) const
344 {
345  if (not isParameter()) {
346  error = "Can not set element " + getPath() + " as it is not a parameter.";
347  return false;
348  }
349 
350  if (asArray != isArray()) {
351 
352  auto toStr = [](bool array) { return array ? "an array" : "a scalar"; };
353  error = "Can not get \"" + getPath() + "\" as " + toStr(asArray) + " because it is " +
354  toStr(isArray());
355  return false;
356  }
357 
358  return true;
359 }

361 // Access validity
362 bool ElementHandle::checkSetValidity(size_t arrayLength, string &error) const
363 {
364  // Settings a parameter necessitates the right to get it
365  if (not checkGetValidity(arrayLength != 0, error)) {
366  return false;
367  }
368 
369  if (!isRogue()) {
370 
371  error = "Can not set parameter \"" + getPath() + "\" as it is not rogue.";
372  return false;
373  }
374 
375  if (arrayLength && (arrayLength != getArrayLength())) {
376 
377  using std::to_string;
378  error = "Array length mismatch for \"" + getPath() + "\", expected: " +
379  to_string(getArrayLength()) + ", got: " + to_string(arrayLength);
380  return false;
381  }
382 
383  return true;
